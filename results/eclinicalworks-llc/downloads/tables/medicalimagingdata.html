EHI Export Schema
Table medicalimagingdata
Description This table stores information about medical imaging data, including patient details, physician details, and file information.
Column DataType Description
id (Primary Key) int Unique identifier for each record in the table. Example: 1, 2, 3.
pid int Patient identifier. Example: 101, 102, 103.
modality varchar Modality of the medical imaging. Example: 'MRI', 'CT', 'X-ray'.
dcm_Patientid_00100020 varchar DICOM Patient id. Example: 'P001', 'P002', 'P003'.
dcm_PatientName_00100010 varchar DICOM Patient Name. Example: 'John Doe', 'Jane Smith', 'Alice Johnson'.
dcm_PerformingPhysicianName_00081050 varchar DICOM Performing Physician Name. Example: 'Dr. Brown', 'Dr. Green', 'Dr. White'.
dcm_PatientDOB_00100030 varchar DICOM Patient Date of Birth. Example: '1980-01-01', '1990-02-15', '2000-12-31'.
dcm_PatientSex_00100040 varchar DICOM Patient Sex. Example: 'male', 'female', 'unknown'.
dcm_RefferingPhysicianName_00080090 varchar DICOM Referring Physician Name. Example: 'Dr. Black', 'Dr. Gray', 'Dr. Blue'.
dcm_RequestingPhysicianName_00321032 varchar DICOM Requesting Physician Name. Example: 'Dr. Red', 'Dr. Yellow', 'Dr. Orange'.
docname varchar Document name. Example: 'report1', 'report2', 'report3'.
dcm_AcquisitionNumber_00200012 varchar DICOM Acquisition Number. Example: 'A001', 'A002', 'A003'.
dcm_Studyid_00200010 varchar DICOM Study id. Example: 'S001', 'S002', 'S003'.
filepath varchar File path of the medical imaging data. Example: '/path/to/file1', '/path/to/file2', '/path/to/file3'.
reviewed char Indicates if the record has been reviewed. Example: 'Y' (Yes), 'N' (No).
moddatetime datetime Date and time when the record was last modified. Example: '2021-01-01 12:34:56', '2021-02-02 23:45:01'.
Deviceid int Device identifier. Example: 1001, 1002, 1003.
facilitycode varchar Facility code. Example: 'FAC001', 'FAC002', 'FAC003'.
reviewedBy int identifier of the user who reviewed the record. Example: 1, 2, 3.
filename varchar Filename of the medical imaging data. Example: 'file1.dcm', 'file2.dcm', 'file3.dcm'.
CreatedAt datetime Date when the record was created. Example: '2021-01-01', '2021-02-02', '2021-03-03'.
