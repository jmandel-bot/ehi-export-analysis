Table inventory_supplier
Description This table stores information about suppliers in the inventory system.
Column DataType Description
id (Primary Key) int Unique identifier for each supplier record. (e.g., 1, 2, 3)
Name varchar Name of the supplier. (e.g., "Supplier A", "Supplier B")
Email varchar Email address of the supplier. (e.g., "supplierA@example.com", "supplierB@example.com")
Phone varchar Phone number of the supplier. (e.g., "+1-123-456-7890", "+1-987-654-3210")
Fax varchar Fax number of the supplier. (e.g., "+1-123-456-7891", "+1-987-654-3211")
Address1 varchar First line of the supplier's address. (e.g., "123 Main St", "456 Elm St")
Address2 varchar Second line of the supplier's address, if applicable. (e.g., "Suite 100", "Apt 4B")
City varchar City where the supplier is located. (e.g., "New York", "Los Angeles")
State varchar State where the supplier is located. (e.g., "NY", "CA")
Country varchar Country where the supplier is located. (e.g., "USA", "Canada")
Zip varchar Postal code of the supplier's address. (e.g., "10001", "90001")
OtherDetails varchar Additional information about the supplier. (e.g., "Preferred supplier", "Discount available")
AccountExecutive varchar Name of the account executive managing the supplier. (e.g., "John Doe", "Jane Smith")
IsDeleted tinyint Flag indicating if the supplier record is deleted (0 for not deleted, 1 for deleted).
UpdatedBy int identifier of the user who last updated the supplier record, foreign key relation to `Supplier_Users` table.
UpdatedOn datetime Date and time when the supplier record was last updated. (e.g., "2021-01-01 12:34:56")
isDefaultSupplier tinyint Flag indicating if the supplier is the default supplier (0 for not default, 1 for default).
contactPerson varchar Name of the contact person at the supplier. (e.g., "Alice Johnson", "Bob Williams")
drugLicenseNo1 varchar First drug license number of the supplier. (e.g., "DL12345", "DL67890")
tin varchar Taxpayer identification Number of the supplier. (e.g., "123-45-6789", "987-65-4321")
vat varchar Value Added Tax number of the supplier. (e.g., "VAT123456", "VAT789012")
corporateid varchar Corporate id of the supplier. (e.g., "Cid123", "Cid456")
contactPersonNum varchar Phone number of the contact person at the supplier. (e.g., "+1-123-456-7892", "+1-987-654-3212")
drugLicenseNo2 varchar Second drug license number of the supplier, if applicable. (e.g., "DL23456", "DL78901")
source tinyint Source of the supplier record (0 for manual entry, 1 for imported, etc.).
isFreeQtyInVatAmt tinyint Flag indicating if free quantity is included in the VAT amount (0 for not included, 1 for included).
AltEmail varchar Alternate email address of the supplier. (e.g., "alt_supplierA@example.com", "alt_supplierB@example.com")
isVision tinyint Flag indicating if the supplier is a vision supplier (0 for not vision, 1 for vision).
isLab tinyint Flag indicating if the supplier is a lab supplier (0 for not lab, 1 for lab).
SupplierNo varchar Supplier number assigned by the inventory system. (e.g., "S123", "S456")
Integrationid int Integration identifier for the supplier, used for external system integration. (e.g., 1001, 1002)
TYPE tinyint Type of the supplier (0 for regular, 1 for special, etc.).
isOrderThruVisionWeb tinyint Flag indicating if orders are placed through the vision web system (0 for not through vision web, 1 for through vision web).
faxDialingCountryid int identifier of the country code for fax dialing, foreign key relation to `countrycodes` table (`id` column).
