Table ptmultiformpayment
Description This table stores information about ePayment transactions, including payment processing responses and related details.
Column DataType Description
id (Primary Key) int Unique identifier for each payment record.
TotalAmount decimal Total amount of the payment transaction, e.g., 100.00.
Discount decimal Discount amount applied to the transaction, e.g., 10.00.
DiscountReason varchar Reason for the discount, e.g., "Promotion".
Cash decimal Cash amount of the payment, e.g., 50.00.
BankCheck decimal Amount paid by bank check, e.g., 25.00.
CheckNumber varchar Bank check number, e.g., "123456".
CheckDate datetime Date of the bank check, e.g., "2022-01-01".
SVC decimal Stored value card (gift card) amount, e.g., 10.00.
SVCNumber varchar Stored value card number, e.g., "1234567890123456".
SVCCCCardVerifyNum varchar Stored value card verification number, e.g., "1234".
CC decimal Credit card payment amount, e.g., 15.00.
CCNumber varchar Credit card number, e.g., "1234567890123456".
CCExp varchar Credit card expiration date, e.g., "012025".
CCType varchar Credit card type, e.g., "Visa".
CCCardbrand varchar Credit card brand, e.g., "MasterCard".
CCCardVerifyNum varchar Credit card verification number, e.g., "123".
magStripeTrack1 varchar Magnetic stripe track 1 data from the credit card.
magStripeTrack2 varchar Magnetic stripe track 2 data from the credit card.
comments varchar Additional comments about the payment, e.g., "Partial payment".
CCName varchar Name on the credit card, e.g., "John Doe".
CCBillingAddress1 varchar Credit card billing address line 1, e.g., "123 Main St".
CCBillingAddress2 varchar Credit card billing address line 2, e.g., "Apt 4B".
CCBillingCity varchar Credit card billing city, e.g., "New York".
CCBillingState varchar Credit card billing state, e.g., "NY".
CCBillingZipCode varchar Credit card billing zip code, e.g., "10001".
CCBillingCountry varchar Credit card billing country, e.g., "USA".
Orderid varchar Order identifier associated with the payment, e.g., "ORD12345".
Bin varchar Bank identification number of the credit card, e.g., "123456".
Merchantid varchar Merchant identifier for the payment processor, e.g., "M12345".
Terminalid varchar Terminal identifier for the payment processor, e.g., "T12345".
TransType char Transaction type, e.g., "CC" or "AC"
IndustryType varchar Industry type for the payment, e.g., "Retail".
PriorAuthCd varchar Prior authorization code for the payment, e.g., "123456".
Userid int User identifier associated with the payment.
encid int Encounter identifier associated with the payment.
facilityid int Facility identifier associated with the payment.
patientid int Patient identifier associated with the payment.
Paymentid int Payment identifier associated with the payment.
DateCreated datetime Date and time the payment record was created, e.g., "2022-01-01 12:34:56".
Resp_txRefNum varchar Payment processor transaction reference number, e.g., "TX123456".
Resp_txRefidx varchar Payment processor transaction reference index, e.g., "01".
Resp_respDateTime varchar Payment processor response date and time, e.g., "20220101123456".
Resp_procStatus varchar Payment processor processing status, e.g., "000000".
Resp_procStatusMessage varchar Payment processor processing status message, e.g., "Approved".
Resp_approvalStatus char Payment processor approval status, e.g., "A".
Resp_respCode char Payment processor response code, e.g., "00".
Resp_avsRespCode char Payment processor address verification response code, e.g., "Y".
Resp_cvvRespCode char Payment processor card verification value response code, e.g., "M".
Resp_authorizationCode varchar Payment processor authorization code, e.g., "123456".
Resp_flexAcctBalance decimal Flexible account balance after the transaction, e.g., 90.00.
Resp_flexAcctPriorBalance decimal Flexible account balance before the transaction, e.g., 100.00.
Resp_flexRedeemedAmt decimal Flexible account redeemed amount, e.g., 10.00.
Resp_flexAcctExpireDate varchar Flexible account expiration date, e.g., "012025".
Resp_batchFailedAcctNum varchar Failed batch account number, e.g., "1234567890123456".
Resp_flexHostTrace varchar Flexible account host trace, e.g., "HT123456".
Resp_retryAttempCount char Retry attempt count, e.g., "02".
Resp_lastRetryDate varchar Last retry date, e.g., "20220101123456".
Resp_FlexRequestedAmount decimal Flexible account requested amount, e.g., 10.00.

Resp_TF_Transid varchar TransFirst transaction identifier, e.g., "TF123456".

Resp_TF_PostedDate varchar TransFirst posted date, e.g., "20220101".
Resp_TF_SettledDate varchar TransFirst settled date, e.g., "20220102".
Resp_TF_AuthCode varchar TransFirst authorization code, e.g., "TF1234".
Resp_TF_Status varchar TransFirst transaction status, e.g., "Approved".
Resp_TF_AVSCode varchar TransFirst address verification code, e.g., "Y".
Resp_TF_Message varchar TransFirst transaction message, e.g., "Transaction approved".
Resp_TF_CVV2Code varchar TransFirst card verification value code, e.g., "M".
TF_Merchantid varchar TransFirst merchant identifier, e.g., "TFM12345".
TF_RegKey varchar TransFirst registration key, e.g., "TFREG12345".
TF_CreditCardURL varchar TransFirst credit card processing URL, e.g., "https://example.com/cc".
TF_CheckVerifyURL varchar TransFirst check verification URL, e.g., "https://example.com/check".
TF_NamespaceURI varchar TransFirst namespace URI, e.g., "http://example.com/ns".
