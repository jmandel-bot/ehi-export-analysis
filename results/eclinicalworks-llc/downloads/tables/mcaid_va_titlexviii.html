Table mcaid_va_titlexviii
Description This table stores information about claim, provider details for Medicaid Virginia Title XVIII form
Column DataType Description
Id (Primary Key) int Unique identifier for each record in the table (e.g., 1, 2, 3).
ClaimNumber int Unique identifier for each claim (e.g., 12345, 67890).
BilProviderNumber varchar Billing provider's unique identification number (e.g., 'AB123456', 'CD789012').
RecipientLastName varchar Last name of the recipient (e.g., 'Smith', 'Johnson').
RecipientFirstName varchar First name of the recipient (e.g., 'John', 'Jane').
RecipientIDNumber varchar Unique identification number for the recipient (e.g., '1234567890', '0987654321').
PatientsAccountNumber varchar Patient's account number (e.g., 'PA1234', 'PA5678').
RenderingProviderNumber varchar Rendering provider's unique identification number (e.g., 'RP123456', 'RP789012').
IsNoOtherCoverage1 char Indicates if there is no other coverage for the first service (e.g., 'Y' for yes, 'N' for no).
IsBilledAndPaid1 char Indicates if the first service is billed and paid (e.g., 'Y' for yes, 'N' for no).
IsBilledAndNoCoverage1 char Indicates if the first service is billed and has no coverage (e.g., 'Y' for yes, 'N' for no).
TypeOfCoverage1 char Type of coverage for the first service (e.g., 'A' for type A, 'B' for type B).
Diagnosis1 varchar Diagnosis code for the first service (e.g., 'D12345', 'D67890').
PlaceOfTreatment1 varchar Place of treatment code for the first service (e.g., 'P123', 'P456').
IsAcc1 char Indicates if the first service is related to an accident (e.g., 'Y' for yes, 'N' for no).
IsEmer1 char Indicates if the first service is related to an emergency (e.g., 'Y' for yes, 'N' for no).
IsOther1 char Indicates if the first service is related to other situations (e.g., 'Y' for yes, 'N' for no).
TypeOfService1 char Type of service code for the first service (e.g., 'S1', 'S2').
ProcedureCode1 varchar Procedure code for the first service (e.g., 'PC123', 'PC456').
NProcedureCode1 varchar Additional procedure code for the first service (e.g., 'NPC123', 'NPC456').
UnitsOfService1 varchar Number of units for the first service (e.g., '1', '2', '3').
AdmDateMM1 char Admission month for the first service (e.g., '01' for January, '12' for December).
AdmDateDD1 char Admission day for the first service (e.g., '01', '31').
AdmDateYY1 char Admission year for the first service (e.g., '20', '21').
BegDateMM1 char Beginning month for the first service (e.g., '01' for January, '12' for December).
BegDateDD1 char Beginning day for the first service (e.g., '01', '31').
BegDateYY1 char Beginning year for the first service (e.g., '20', '21').
EndDateMM1 char Ending month for the first service (e.g., '01' for January, '12' for December).
EndDateDD1 char Ending day for the first service (e.g., '01', '31').
EndDateYY1 char Ending year for the first service (e.g., '20', '21').
ChargesDec1 varchar Decimal charges for the first service (e.g., '123.45', '678.90').
ChargesUnit1 char Unit charges for the first service (e.g., '01', '02').
AllowedDec1 varchar Decimal allowed amount for the first service (e.g., '123.45', '678.90').
AllowedUnit1 char Unit allowed amount for the first service (e.g., '01', '02').
MedicarePayDec1 varchar Decimal Medicare payment for the first service (e.g., '123.45', '678.90').
MedicarePayUnit1 char Unit Medicare payment for the first service (e.g., '01', '02').
DeductDec1 varchar Decimal deductible amount for the first service (e.g., '123.45', '678.90').
DeductUnit1 char Unit deductible amount for the first service (e.g., '01', '02').
CoInsDec1 varchar Decimal coinsurance amount for the first service (e.g., '123.45', '678.90').
CoInsUnit1 char Unit coinsurance amount for the first service (e.g., '01', '02').
PaidOCarrierDec1 varchar Decimal amount paid by other carrier for the first service (e.g., '123.45', '678.90').
PaidOCarrierUnit1 char Unit amount paid by other carrier for the first service (e.g., '01', '02').
LtcPayDec1 varchar Decimal long-term care payment for the first service (e.g., '123.45', '678.90').
LtcPayUnit1 char Unit long-term care payment for the first service (e.g., '01', '02').
Ndc1 varchar National Drug Code for the first service (e.g., 'NDC12345', 'NDC67890').
