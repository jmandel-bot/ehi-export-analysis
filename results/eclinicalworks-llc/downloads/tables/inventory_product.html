Table inventory_product
Description This table stores information about products in the inventory, including their details, quantities, prices, and taxes.
Column DataType Description
id (Primary Key) int Unique identifier for each product in the inventory.
Name varchar Name of the product.
Description text Description of the product.
Barcode varchar Barcode of the product.
Category_id int identifier for the category the product belongs to. Foreign key relation to the `Category` table.
IsDiscontinued tinyint Flag indicating whether the product is discontinued (1) or not (0).
OrderQtyUnit varchar Unit of measurement for the order quantity, e.g., "Box(es)".
OrderQty decimal Quantity of the product ordered.
UnitQtyUnit varchar Unit of measurement for the unit quantity, e.g., "Vial(s)".
UnitQty decimal Quantity of the product in units.
UsageQtyUnit varchar Unit of measurement for the usage quantity, e.g., "CC(s)".
HasSalesTax tinyint Flag indicating whether the product has sales tax (1) or not (0).
HasCosmeticTax tinyint Flag indicating whether the product has cosmetic tax (1) or not (0).
HasCustomTax tinyint Flag indicating whether the product has custom tax (1) or not (0).
CPT_id int identifier for the Current Procedural Terminology (CPT) code. Foreign key relation to the `CPT` table.
CPT_Code varchar CPT code of the product.
CPT_Description varchar Description of the CPT code.
CPT_Price decimal Price of the product according to the CPT code.
HasSerialNo tinyint Flag indicating whether the product has a serial number (1) or not (0).
HasExpiryDate tinyint Flag indicating whether the product has an expiry date (1) or not (0).
SalesOrderPrice decimal Sales price of the product when ordered.
SalesUnitPrice decimal Sales price of the product per unit.
SalesUsagePrice decimal Sales price of the product based on usage.
AlertForOrderQty decimal Alert threshold for the order quantity.
AlertForUnitQty decimal Alert threshold for the unit quantity.
AlertForUsageQty decimal Alert threshold for the usage quantity.
DefaultOrderQty decimal Default order quantity for the product.
DefaultUnitQty decimal Default unit quantity for the product.
DefaultUsageQty decimal Default usage quantity for the product.
AlertBeforeExpiryDate int Number of days before the expiry date to trigger an alert.
IsDeleted tinyint Flag indicating whether the product is deleted (1) or not (0).
UpdatedBy int identifier for the user who last updated the product information. Foreign key relation to the `Users` table.
UpdatedOn datetime time stamp of the last update on the product information.
Manufacturer_id int identifier for the manufacturer of the product. Foreign key relation to the `Manufacturer` table.
ImgSrc varchar Image source URL for the product.
Manufacturer varchar Name of the manufacturer.
itemid int identifier for the item.
catid int identifier for the category.
IsRx tinyint Flag indicating whether the product is a prescription item (1) or not (0).
Rx_CatItemid int identifier for the prescription category item.
Rx_CatItemName varchar Name of the prescription category item.
Rx_Itemid int identifier for the prescription item.
Rx_ItemName varchar Name of the prescription item.
Rx_Dosage varchar Dosage of the prescription item.
Rx_NDCCode varchar National Drug Code (NDC) for the prescription item.
isLabsReports tinyint Flag indicating whether the product is related to lab reports (1) or not (0).
labsReports_id int identifier for the lab report. Foreign key relation to the `LabsReports` table.
generic varchar Generic name of the product.
Rx_Strength varchar Strength of the prescription item.
Rx_Formulation varchar Formulation of the prescription item.
source tinyint Source of the product information.
Currency varchar Currency for the product prices, e.g., "USD".
isDiscountAllowed tinyint Flag indicating whether a discount is allowed on the product (1) or not (0).
feeSchid int identifier for the fee schedule.
createdBy int identifier for the user who created the product entry. Foreign key relation to the `Users` table.
createdOn datetime time stamp of when the product entry was created.
isChargeable tinyint Flag indicating whether the product is chargeable (1) or not (0).
externalCode varchar External code for the product.
isBySalesPrice tinyint Flag indicating whether the product is priced by sales price (1) or not (0).
isFromInterface tinyint Flag indicating whether the product is from an interface (1) or not (0).
cptFeeSchids varchar Comma-separated list of CPT fee schedule identifiers.
hubid varchar identifier for the hub.
item_number varchar Item number of the product.
model_number varchar Model number of the product.
stock_type tinyint Stock type of the product.
isnondisposable tinyint Flag indicating whether the product is non-disposable (1) or not (0).
UPC_Code varchar Universal Product Code (UPC) for the product.
modality int Modality of the product.
uniqueKey tinyint Flag indicating whether the product has a unique key (1) or not (0).
uniqueCode varchar Unique code for the product.
prodCustomField1 varchar Custom field 1 for the product.
prodCustomField2 varchar Custom field 2 for the product.
prodCustomField3 varchar Custom field 3 for the product.
prodCustomField4 varchar Custom field 4 for the product.
prodCustomField5 varchar Custom field 5 for the product.
country varchar Country of origin for the product.
styleid int identifier for the style of the product.
type int Type of the product.
isTrial tinyint Flag indicating whether the product is a trial item (1) or not (0).
modalityid int identifier for the modality of the product.
v_sku varchar SKU (Stock Keeping Unit) of the product.
haslatex tinyint Flag indicating whether the product contains latex (1) or not (0).
hasimplant tinyint Flag indicating whether the product is an implant (1) or not (0).
isService tinyint Flag indicating whether the product is a service (1) or not (0).
hasHazardousMaterial tinyint Flag indicating whether the product contains hazardous material (1) or not (0).
ptChargeableItemNumber varchar Chargeable item number for the product.
ptNonbillableExpense varchar Non-billable expense for the product.
ptNonbillableExpenseDescription varchar Description of the non-billable expense for the product.
ptBillableExpense varchar Billable expense for the product.
ptBillableExpenseDescription varchar Description of the billable expense for the product.
HasLotNo tinyint Flag indicating whether the product has a lot number (1) or not (0).
rx_class varchar Class of the prescription item.
brand_id int identifier for the brand of the product.
endDate datetime End date for the product.
eposDescription varchar EPOS (Electronic Point of Sale) description for the product.
iscustom tinyint Flag indicating whether the product is custom (1) or not (0).
Lifecycle_Status_id int identifier for the lifecycle status of the product.
retailDept_id int identifier for the retail department of the product.
hasTaxExempt tinyint Flag indicating whether the product is tax-exempt (1) or not (0).
vat_code varchar Value-added tax (VAT) code for the product.
costprice decimal Cost price of the product.
contactlensType int Type of contact lens for the product.
tempProductid int Temporary product identifier.
lastsyncdate datetime time stamp of the last synchronization of the product data.
omtlensprice tinyint Flag indicating whether the product has an OMT lens price (1) or not (0).
valid varchar Value identifier for the product.
concentration varchar Concentration of the product.
rx_packsize decimal Pack size of the prescription item.
rx_packageSizeUnitDesc varchar Unit description for the package size of the prescription item.
rx_secondaryNDC varchar Secondary National Drug Code (NDC) for the prescription item.
rx_packsizeQty decimal Pack size quantity of the prescription item.
isManualyAddedNDC tinyint Flag indicating whether the NDC is manually added (1) or not (0).
