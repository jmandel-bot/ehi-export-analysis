Table vision_cmnprint
Description This table stores information related to vision prescriptions, including patient and provider details, store information, and prescription data.
Column DataType Description
id (Primary Key) int Unique identifier for each record in the table. (e.g., 1, 2, 3)
Encid int Unique identifier for the encounter associated with the prescription. (e.g., 101, 102, 103)
Insid int Unique identifier for the insurance associated with the prescription, can be NULL. (e.g., 201, 202, NULL)
Storeid varchar Store identifier where the prescription was made, can be NULL. (e.g., "STR001", "STR002", NULL)
StoreAddress varchar Store address where the prescription was made, can be NULL. (e.g., "123 Main St", "456 Oak St", NULL)
StoreCityStateZip varchar Store city, state, and zip code where the prescription was made, can be NULL. (e.g., "New York, NY 10001", "Los Angeles, CA 90001", NULL)
StorePhoneNo varchar Store phone number where the prescription was made, can be NULL. (e.g., "555-123-4567", "555-987-6543", NULL)
StoreFax varchar Store fax number where the prescription was made, can be NULL. (e.g., "555-123-4568", "555-987-6544", NULL)
OrderNo varchar Order number for the prescription, can be NULL. (e.g., "ORD001", "ORD002", NULL)
PtName varchar Patient's full name, can be NULL. (e.g., "John Doe", "Jane Smith", NULL)
PtDOB varchar Patient's date of birth, can be NULL. (e.g., "01/01/1980", "12/31/1990", NULL)
PtAddress varchar Patient's address, can be NULL. (e.g., "789 Maple St", "321 Elm St", NULL)
PtCityStateZip varchar Patient's city, state, and zip code, can be NULL. (e.g., "Chicago, IL 60601", "Houston, TX 77001", NULL)
PtPhoneNo varchar Patient's phone number, can be NULL. (e.g., "555-123-7890", "555-987-3210", NULL)
PrName varchar Provider's full name, can be NULL. (e.g., "Dr. Alice Johnson", "Dr. Bob Williams", NULL)
PrAddress varchar Provider's address, can be NULL. (e.g., "1234 Clinic St", "5678 Hospital St", NULL)
PrCityStateZip varchar Provider's city, state, and zip code, can be NULL. (e.g., "San Francisco, CA 94101", "Miami, FL 33101", NULL)
PrFax varchar Provider's fax number, can be NULL. (e.g., "555-123-1111", "555-987-2222", NULL)
PrPhoneNo varchar Provider's phone number, can be NULL. (e.g., "555-123-3333", "555-987-4444", NULL)
HICNumber varchar Health Insurance Claim Number, can be NULL. (e.g., "HIC123456", "HIC789012", NULL)
NPI varchar National Provider identifier, can be NULL. (e.g., "NPI1234567890", "NPI0987654321", NULL)
PrescriptionDate varchar Date when the prescription was made, can be NULL. (e.g., "01/01/2022", "02/15/2022", NULL)
DateOfService varchar Date of service for the prescription, can be NULL. (e.g., "01/02/2022", "02/16/2022", NULL)
DateOfCataractSurgery varchar Date of cataract surgery, if applicable, can be NULL. (e.g., "12/01/2021", "11/15/2021", NULL)
EyeType varchar Type of eye for the prescription (e.g., "Right", "Left", "Both")
RightSphere varchar Sphere value for the right eye, can be NULL. (e.g., "+1.00", "-1.50", NULL)
RightCylinder varchar Cylinder value for the right eye, can be NULL. (e.g., "+0.50", "-0.75", NULL)
RightAxis varchar Axis value for the right eye, can be NULL. (e.g., "180", "90", NULL)
RightAdd varchar Add value for the right eye, can be NULL. (e.g., "+2.00", "+1.50", NULL)
RightPrism varchar Prism value for the right eye, can be NULL. (e.g., "2", "3", NULL)
LeftSphere varchar Sphere value for the left eye, can be NULL. (e.g., "+1.25", "-1.75", NULL)
LeftCylinder varchar Cylinder value for the left eye, can be NULL. (e.g., "+0.75", "-0.50", NULL)
LeftAxis varchar Axis value for the left eye, can be NULL. (e.g., "180", "90", NULL)
LeftAdd varchar Add value for the left eye, can be NULL. (e.g., "+2.50", "+1.25", NULL)
LeftPrism varchar Prism value for the left eye, can be NULL. (e.g., "2", "3", NULL)
ICDCode1 varchar ICD (International Classification of Diseases) codes, can be NULL. (e.g., "H52.4", "H25.1", NULL)
ICDCodeDesc1 varchar Descriptions for the ICD codes, can be NULL. (e.g., "Astigmatism", "Cataract", NULL)
CurrentDate varchar Current date when the record was created, can be NULL. (e.g., "03/01/2022", "03/15/2022", NULL)
ICDCode10 varchar ICD (International Classification of Diseases) codes, can be NULL. (e.g., "H52.4", "H25.1", NULL)
