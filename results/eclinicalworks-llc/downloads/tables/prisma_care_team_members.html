Table prisma_care_team_members
Description This table stores information about the members of a care team in a healthcare organization.
Column DataType Description
id (Primary Key) bigint Unique identifier for each care team member. (e.g., 1, 2, 3)
sectionId bigint Identifier for the section the care team member belongs to. Foreign key relation to `prisma_ccd_document_section.id`. (e.g., 101, 102, 103)
firstName varchar First name of the care team member. (e.g., John, Jane, Alice)
lastName varchar Last name of the care team member. (e.g., Smith, Doe, Johnson)
middleName varchar Middle name of the care team member. (e.g., Michael, Marie, Elizabeth)
suffix varchar Suffix of the care team member's name. (e.g., Jr., Sr., III)
previousName varchar Previous name of the care team member, if applicable. (e.g., Jane Brown, Alice Johnson)
addressLine1 varchar First line of the care team member's address. (e.g., 123 Main St)
addressLine2 varchar Second line of the care team member's address, if applicable. (e.g., Apt 4B, Suite 200)
city varchar City of the care team member's address. (e.g., New York, Los Angeles, Chicago)
state varchar State of the care team member's address. (e.g., NY, CA, IL)
postalCode varchar Postal code of the care team member's address. (e.g., 10001, 90001, 60601)
country varchar Country of the care team member's address. (e.g., United States, Canada, United Kingdom)
contactType varchar Type of contact information for the care team member. (e.g., phone, email)
contactNo varchar Contact number or email address of the care team member. (e.g., 555-123-4567, john.doe@example.com)
participant_role varchar Role of the care team member in the team. (e.g., physician, nurse, therapist)
participant_member_reference int Reference number for the care team member. (e.g., 1001, 1002, 1003)
participant_member_display varchar Display name for the care team member. (e.g., Dr. John Smith, Nurse Jane Doe)
participant_period_start datetime Start date of the care team member's participation in the team. (e.g., 2021-01-01 00:00:00)
participant_period_end datetime End date of the care team member's participation in the team, if applicable. (e.g., 2021-12-31 00:00:00)
care_team_id varchar Identifier for the care team the member belongs to. (e.g., CT001, CT002, CT003)
care_team_name varchar Name of the care team the member belongs to. (e.g., Cardiology Team, Oncology Team, Pediatrics Team)
care_team_status varchar Status of the care team the member belongs to. (e.g., active, inactive, completed)
care_team_mem_start_date datetime Start date of the care team member's membership in the team. (e.g., 2021-01-01 00:00:00)
care_team_mem_end_date datetime End date of the care team member's membership in the team, if applicable. (e.g., 2021-12-31 00:00:00)
fhir_id int Key to store id of fhir_source_table
isException tinyint Flag indicating if there is an exception related to the care team member (1 for true, 0 for false).
exceptionMessage varchar Message describing the exception, if any. Empty if there is no exception.
sourcePriority int Priority of the data source for the care team member. (e.g., 1, 2, 3)
duplicatedBy int Identifier for the care team member that this record is duplicated by. (e.g., -1, 1, 2)
