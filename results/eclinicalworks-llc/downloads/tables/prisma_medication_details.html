Table prisma_medication_details
Description This table stores information about medication details for patients, including medication name, dosage, frequency, and other related data.
Column DataType Description
id (Primary Key) bigint Unique identifier for each medication detail record. Example: 1, 2, 3
sectionId bigint Foreign key referring to the `prisma_ccd_document_section` table's `id` column. It represents the section in which the medication detail is present. Example: 101, 102, 103
medicationname varchar Name of the medication. Example: "Aspirin", "Ibuprofen", "Paracetamol"
moodCode varchar Code representing the mood of the medication. Example: "EVN", "INT", "RQO"
startDate varchar Start date of the medication in a string format. Example: "2021-01-01", "2021-06-15"
enddate varchar End date of the medication in a string format. Example: "2021-12-31", "2021-07-30"
frequencyValue varchar Frequency value of the medication. Example: "1", "2", "3"
frequencyUnit varchar Unit of the frequency value. Example: "day", "week", "month"
operator char Operator for the medication. Example: "A"
institutionSpecified tinyint Tiny integer indicating if the institution specified the medication. Example: 0, 1
routeCode varchar Code representing the route of administration for the medication. Example: "PO", "IM", "IV"
routeName varchar Name of the route of administration. Example: "Oral", "Intramuscular", "Intravenous"
siteCode varchar Code representing the site of administration for the medication. Example: "LA", "RA", "LL"
siteName varchar Name of the site of administration. Example: "Left Arm", "Right Arm", "Left Leg"
doseQuantity varchar Quantity of the medication dose. Example: "100", "200", "500"
doseUnit varchar Unit of the medication dose. Example: "mg", "g", "ml"
rateQuantity varchar Quantity of the medication rate. Example: "10", "20", "50"
rateUnit varchar Unit of the medication rate. Example: "mg/min", "ml/h", "g/day"
instruction text Long text containing instructions for the medication. Example: "Take with food", "Do not crush or chew"
ndcCode varchar National Drug Code for the medication. Example: "12345-6789", "98765-4321"
ndcName varchar Name of the National Drug Code. Example: "Aspirin 81mg", "Ibuprofen 200mg"
mfgCompany varchar Name of the manufacturing company for the medication. Example: "Pfizer", "Johnson & Johnson"
statusCode varchar Status code of the medication. Example: "active", "completed", "suspended"
rxNormCode varchar RxNorm code for the medication. Example: "198440", "313782"
rxnormname varchar Name of the RxNorm code. Example: "Aspirin 81mg oral tablet", "Ibuprofen 200mg oral tablet"
isException tinyint Tiny integer indicating if the medication is an exception. Example: 0, 1
productNameText varchar Text of the product name. Example: "Aspirin 81mg", "Ibuprofen 200mg"
titleMedsName varchar Title of the medication name. Example: "Aspirin", "Ibuprofen"
strengthValue varchar Strength value of the medication. Example: "81", "200"
strengthFromText varchar Strength value from the text. Example: "81mg", "200mg"
lastFill varchar Last fill date of the medication in a string format. Example: "2021-01-01", "2021-06-15"
sourcePriority int Integer representing the priority of the source. Example: 1, 2, 3
codeSystemPriority int Integer representing the priority of the code system. Example: 1, 2, 3
duplicatedBy int Identifier for the record that duplicates this record. A value of -1 indicates no duplication. (e.g., -1, 5, 10)
exceptionMessage varchar Comma separated ids, each id pointing to a message in table "prisma_ccd_preprocessor_error_messages". Empty if there is no exception.
prescriberFName varchar First name of the prescriber. Example: "John", "Jane"
prescriberLName varchar Last name of the prescriber. Example: "Doe", "Smith"
encounterId int Integer representing the encounter ID. Example: 1001, 1002, 1003
requesterId int Integer representing the requester ID. Example: 1, 2, 3
