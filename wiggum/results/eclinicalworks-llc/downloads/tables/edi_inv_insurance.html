Table edi_inv_insurance
Description This table stores information about insurance claims related to invoices.
Column DataType Description
id (Primary Key) int Unique identifier for each record in the table.
Invoiceid int identifier for the invoice related to the insurance claim. Foreign key referring to the `id` column in the `edi_invoice` table.
SeqNo int Sequence number for the insurance (1, 2, etc)
Payorid varchar Identifier for the payer of the insurance claim.
Insid int Identifier for the insurance company. Foreign key referring to the id column in the Insurance table.
groupNo varchar Group number for the insurance policy.
subscriberNo varchar Subscriber number for the insurance policy.
InsType varchar Type of insurance (e.g., PPO, HMO, etc.).
FileStatus varchar Status of the insurance claim file (e.g., pending, approved, etc.).
InsOrder smallint Order of the insurance claim in the processing sequence.
ClaimIndicator varchar Indicator for the type of claim (e.g., primary, secondary, etc.).
PaymentSource varchar Source of the payment for the insurance claim.
PayerClaimOfficeNo varchar Office number of the payer for the insurance claim.
PPOIndex varchar Index number for the preferred provider organization (PPO) related to the insurance claim.
PPOid varchar identifier for the preferred provider organization (PPO) related to the insurance claim.
priorauthno varchar Prior authorization number for the insurance claim.
PtSigSource varchar Source of the patient's signature for the insurance claim.
groupName varchar Name of the group associated with the insurance policy.
AssignBenefits varchar Indicator for the assignment of benefits (e.g., yes, no, etc.).
Grid int Guarantor id for Guarantor associated with claim invoice
GrRel int Relation with Guarantor
IsGrPt int Indicator for whether the guarantoris is a patient (e.g., 1 for yes, 0 for no).
deleteFlag int Indicator for whether the record has been deleted (e.g., 1 for yes, 0 for no).
displayIndex int Display index for the insurance claim record.
SuppInsInd char Supplementary insurance indicator.
Medicaidid varchar identifier for the Medicaid insurance claim.
patSubscriberNo varchar Patient subscriber number for the insurance policy.
KenPAC int KenPAC number for the insurance claim.
InsuranceClass varchar Class of the insurance policy (e.g., individual, family, etc.).
ReferralNumber varchar This column refers to the ReferralNumber column in the Referral table.
ServAuthExCode char Service authorization exception code for the insurance claim.
OrgClaimIndicator varchar Original claim indicator for the insurance claim.
InsTypeOrg varchar Original insurance type for the insurance claim.
InsTypeTransitionStage tinyint Transition stage for the insurance type (e.g., 0 for no transition, 1 for transition).
InsuredAltLName varchar Alternate last name of the insured person.
InsuredAltFName varchar Alternate first name of the insured person.
InsuredAltMiddleInitial varchar Alternate middle initial of the insured person.
SubscriberSecondaryidentifier varchar Secondary identifier for the subscriber of the insurance policy.
SubscriberidentificationQualifier varchar Qualifier for the subscriber identification.
PatientAltLName varchar Alternate last name of the patient.
PatientAltFName varchar Alternate first name of the patient.
PatientAltMiddleInitial varchar Alternate middle initial of the patient.
SubscriberSecondaryInvRef varchar Secondary invoice reference for the subscriber.
SubscriberSecondaryInvRefQualifier varchar Qualifier for the secondary invoice reference.
SecondaryInsuranceTransitionDelete tinyint Indicator for whether the secondary insurance transition has been deleted (e.g., 1 for yes, 0 for no).
EAFAmt decimal Electronic Funds Transfer (EFT) amount for the insurance claim.
PropertyCasualtyClaimNumber varchar Property and casualty claim number for the insurance claim.
cobamt_totalnoncovered decimal Total non-covered amount for the coordination of benefits (COB) of the insurance claim.
casualtycontactname varchar Contact name for the casualty insurance claim.
casualtyphoneno varchar Phone number for the casualty insurance claim contact.
casualtyextn varchar Extension number for the casualty insurance claim contact.
