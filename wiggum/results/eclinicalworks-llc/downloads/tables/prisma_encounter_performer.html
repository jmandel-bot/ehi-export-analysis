Table prisma_encounter_performer
Description This table stores information about performers involved in medical encounters, including their personal details, contact information, and associated organization details.
Column DataType Description
id (Primary Key) bigint Unique identifier for each performer record. Example: 1, 2, 3.
encounterDetailId bigint Foreign key relation to the `prisma_encounter_details` table, representing the encounter this performer is associated with. Example: 101, 102, 103.
userId varchar Unique identifier for the user associated with the performer. Example: "U123", "U456", "U789".
addressLine1 varchar First line of the performer's address. Example: "123 Main St", "456 Oak St".
addressLine2 varchar Second line of the performer's address (optional). Example: "Apt 4B", "Suite 300".
city varchar City where the performer is located. Example: "New York", "Los Angeles".
county varchar County where the performer is located. Example: "Los Angeles County", "Cook County".
state varchar State where the performer is located. Example: "CA", "NY", "TX".
postalcode varchar Postal code of the performer's address. Example: "90001", "10001".
country varchar Country where the performer is located. Example: "USA", "Canada".
homePhone varchar Home phone number of the performer. Example: "(555) 123-4567", "(555) 987-6543".
workPhone varchar Work phone number of the performer. Example: "(555) 234-5678", "(555) 876-5432".
faxnumber varchar Fax number of the performer. Example: "(555) 345-6789", "(555) 765-4321".
personFName varchar First name of the performer. Example: "John", "Jane".
personLName varchar Last name of the performer. Example: "Doe", "Smith".
personSufix varchar Suffix of the performer's name (optional). Example: "Jr.", "III".
organizationId varchar Unique identifier for the organization the performer is associated with. Example: "O123", "O456", "O789".
organizationname varchar Name of the organization the performer is associated with. Example: "General Hospital", "City Clinic".
organizationPhone varchar Phone number of the organization. Example: "(555) 456-7890", "(555) 654-3210".
organizationFax varchar Fax number of the organization. Example: "(555) 567-8901", "(555) 543-2109".
organizationAddLine1 varchar First line of the organization's address. Example: "789 Main St", "321 Oak St".
organizationAddLine2 varchar Second line of the organization's address (optional). Example: "Floor 2", "Building B".
organizationCity varchar City where the organization is located. Example: "San Francisco", "Chicago".
organizationState varchar State where the organization is located. Example: "IL", "FL", "WA".
organizationPostalCode varchar Postal code of the organization's address. Example: "94102", "60601".
organizationCountry varchar Country where the organization is located. Example: "USA", "UK".
