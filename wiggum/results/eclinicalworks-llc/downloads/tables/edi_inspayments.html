EHI Export Schema
Table edi_inspayments
Description This table stores information about insurance payments made to a provider.
Column DataType Description
paymentId (Primary Key) int Unique identifier for each payment. (e.g., 1, 2, 3)
PayorType smallint Type of payor making the payment. (e.g., 0, 1, 2)
payorId int Unique identifier for the payor. (e.g., 101, 102, 103)
providerId int Unique identifier for the provider receiving the payment. (e.g., 201, 202, 203)
amount decimal Amount of the payment. (e.g., 100.00, 200.00, 300.00)
checkNo varchar Check number associated with the payment. (e.g., '123456', '789012', '345678')
type varchar Type of payment. (e.g., 'Cash', 'Check', 'Credit Card')
date datetime Date of the payment. (e.g., '2021-01-01', '2021-02-01', '2021-03-01')
checkDate datetime Date on the check. (e.g., '2021-01-01', '2021-02-01', '2021-03-01')
memo text Additional information about the payment. (e.g., 'Payment for service A', 'Payment for service B')
PostedBy int Unique identifier for the user who posted the payment. (e.g., 301, 302, 303)
PostedDt datetime Date when the payment was posted. (e.g., '2021-01-01', '2021-02-01', '2021-03-01')
PostedTime datetime Time when the payment was posted. (e.g., '12:00:00', '13:00:00', '14:00:00')
modifiedDate datetime time stamp of the last modification to the payment record. (e.g., '2021-01-01 12:00:00', '2021-02-01 13:00:00')
facilityId int Unique identifier for the facility where the payment was made. (e.g., 401, 402, 403)
paymentLock int Indicates if the payment is locked. (e.g., 0, 1)
lockedBy int Unique identifier for the user who locked the payment. (e.g., 301, 302, 303)
deleteFlag smallint Indicates if the payment record is deleted. (e.g., 0, 1)
UnpostedAmount decimal Unposted amount of the payment. (e.g., 50.00, 100.00, 150.00)
PaymentCode varchar Code associated with the payment. (e.g., 'PMT01', 'PMT02', 'PMT03')
EOMB1 varchar Additional information about the payment. (e.g., 'EOMB1A', 'EOMB1B', 'EOMB1C')
creditCardType int Type of credit card used for the payment. (e.g., 1, 2, 3)
closingId int Unique identifier for the closing associated with the payment. (e.g., 501, 502, 503)
UnpostedCPT decimal Unposted CPT (Current Procedural Terminology) amount of the payment. (e.g., 25.00, 50.00, 75.00)
checkTaxAmt decimal Tax amount on the check. (e.g., 5.00, 10.00, 15.00)
checkTotalAmt decimal Total amount on the check. (e.g., 105.00, 210.00, 315.00)
eraDetailId int Unique identifier for the ERA (Electronic Remittance Advice) detail associated with the payment. (e.g., 601, 602, 603)
BatchId int Unique identifier for the batch associated with the payment. (e.g., 701, 702, 703)
ParentPaymentId int Unique identifier for the parent payment. (e.g., 1, 2, 3)
PracticeId int Unique identifier for the practice associated with the payment. (e.g., 801, 802, 803)
VoidFlag tinyint Indicates if the payment is void. (e.g., 0, 1)
RecreatedFlag tinyint Indicates if the payment has been recreated. (e.g., 0, 1)
transid varchar Transaction identifier for the payment. (e.g., 'TRANS001', 'TRANS002', 'TRANS003')
depositDate datetime Date of the deposit associated with the payment. (e.g., '2021-01-01', '2021-02-01', '2021-03-01')
voidPayment int Unique identifier for the void payment. (e.g., 1, 2, 3)
voidId int Unique identifier for the void payment. (e.g., 1, 2, 3)
isCollectionPayment tinyint Indicates if the payment is a collection payment. (e.g., 0, 1)
isreviewed smallint Indicates if the payment has been reviewed. (e.g., -1, 0, 1)
encid int Unique identifier for the encounter associated with the payment. (e.g., 901, 902, 903)
isexception tinyint Indicates if the payment is an exception. (e.g., 0, 1)
createdFromAppt tinyint Indicates if the payment was created from an appointment. (e.g., 0, 1)
