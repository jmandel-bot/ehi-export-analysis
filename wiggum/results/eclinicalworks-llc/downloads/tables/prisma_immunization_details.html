Table prisma_immunization_details
Description This table stores immunization details for patients, including vaccine information, administration details, and any exceptions or additional notes.
Column DataType Description
id (Primary Key) bigint Unique identifier for each immunization detail record. (e.g., 1, 2, 3)
sectionId bigint Identifier for the section this immunization detail belongs to. Foreign key relation to `prisma_ccd_document_section.id`.
entryId varchar Unique identifier for the entry in the immunization detail. (e.g., 'entry1', 'entry2')
vaccineName varchar Name of the vaccine administered. (e.g., 'Pfizer COVID-19 Vaccine', 'Moderna COVID-19 Vaccine')
administrationTime varchar Time of vaccine administration in a string format. (e.g., '2021-08-01 10:30:00')
statusCode varchar Status code of the immunization. (e.g., 'completed', 'not-done')
cvxCode varchar CVX code for the vaccine product. (e.g., '207', '208')
cvxName varchar CVX name for the vaccine product. (e.g., 'COVID-19, mRNA, LNP-S, PFIZER-BIONTECH', 'COVID-19, mRNA, LNP-S, MODERNA')
routeCode varchar Code for the route of administration. (e.g., 'IM', 'SC')
routeName varchar Name of the route of administration. (e.g., 'Intramuscular', 'Subcutaneous')
lotNumber varchar Lot number of the vaccine product. (e.g., 'AB12345', 'CD67890')
manufacturedOrgName varchar Name of the organization that manufactured the vaccine. (e.g., 'Pfizer', 'Moderna')
rejectionReasonCode varchar Code for the reason the vaccine was rejected. (e.g., '01', '02')
rejectionReason varchar Description of the reason the vaccine was rejected. (e.g., 'Patient refusal', 'Medical contraindication')
productNameCodingSystem varchar Coding system used for the vaccine product name. (e.g., 'CVX', 'NDC')
productNameCodeVersion varchar Version of the coding system used for the vaccine product name. (e.g., '1.0', '2.0')
productNameCodeValue varchar Code value of the vaccine product name in the coding system. (e.g., '207', '208')
productNameCvxCodingSystem varchar Coding system used for the CVX vaccine product name. (e.g., 'CVX')
productNameCvxCodeVersion varchar Version of the CVX coding system used for the vaccine product name. (e.g., '1.0', '2.0')
productNameBrandName varchar Brand name of the vaccine product. (e.g., 'Pfizer-BioNTech COVID-19 Vaccine', 'Moderna COVID-19 Vaccine')
notAdminReasonOther varchar Other reasons for not administering the vaccine. (e.g., 'Patient not eligible', 'Vaccine not available')
notAdminReasonDescription varchar Description of the reason for not administering the vaccine. (e.g., 'Patient has a history of allergic reactions', 'Vaccine supply shortage')
notAdminReasonCodingSystem varchar Coding system used for the reason for not administering the vaccine. (e.g., 'SNOMED-CT', 'ICD-10')
notAdminReasonCoding varchar Code value of the reason for not administering the vaccine in the coding system. (e.g., '123456', '789012')
sourcePriority int Priority of the source for the immunization detail. (e.g., 1, 2, 3)
codeSystemPriority int Priority of the coding system for the immunization detail. (e.g., 1, 2, 3)
duplicatedBy int Identifier for the record that duplicates this immunization detail. (e.g., -1, 1, 2)
statusReason varchar Reason for the immunization status. (e.g., 'Patient refusal', 'Medical contraindication')
statusReasonText varchar Text description of the reason for the immunization status. (e.g., 'Patient refused due to fear of side effects', 'Patient has a history of allergic reactions')
administrationTimeString varchar Time of vaccine administration in a string format. (e.g., '2021-08-01 10:30:00')
doseQuantityValue varchar Quantity value of the vaccine dose. (e.g., '0.5', '1.0')
doseQuantityUnit varchar Unit of the vaccine dose quantity. (e.g., 'mL', 'mg')
encounterId int Identifier for the encounter associated with the immunization detail. (e.g., 1, 2, 3)
expirationDate datetime Expiration date of the vaccine product. (e.g., '2022-12-31 00:00:00')
additionalNotes varchar Additional notes related to the immunization detail. (e.g., 'Patient experienced mild side effects', 'Vaccine administered by Nurse Jane Doe')
siteCode varchar Code for the site where the vaccine was administered. (e.g., 'LA', 'RA')
siteSystem varchar Coding system used for the site where the vaccine was administered. (e.g., 'SNOMED-CT', 'ICD-10')
siteDisplay varchar Display name of the site where the vaccine was administered. (e.g., 'Left arm', 'Right arm')
isException tinyint Flag indicating if there was an exception in the immunization detail (0 for no exception, 1 for exception). (e.g., 0, 1)
exceptionMessage varchar Message describing the exception, if any. Empty if there is no exception.
