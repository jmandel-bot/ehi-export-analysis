Table tbladditionalinfodecrypted
Description This table stores additional decrypted information about patients, their contact details, employment details, and preferred pharmacy.
Column DataType Description
id (Primary Key) int Unique identifier for each record in the table (e.g., 1, 2, 3).
pid int Patient identifier, foreign key relation to the patient's table (e.g., 101, 102, 103).
mobileno varchar Mobile phone number of the patient (e.g., '123-456-7890', '987-654-3210').
workPhone varchar Work phone number of the patient (e.g., '555-123-4567', '555-987-6543').
Pcpid int Primary care provider identifier, foreign key relation to the provider's table (e.g., 1, 2, 3).
RefPrid int Referring provider identifier, foreign key relation to the provider's table (e.g., 4, 5, 6).
RendPrid int Rendering provider identifier, foreign key relation to the provider's table (e.g., 7, 8, 9).
maritalStatus varchar Marital status of the patient (e.g., 'Single', 'Married', 'Divorced').
strAddress1 varchar Street address line 1 of the patient (e.g., '123 Main St', '456 Elm St').
strAddress2 varchar Street address line 2 of the patient (e.g., 'Apt 4B', 'Suite 200').
city varchar City of the patient's address (e.g., 'New York', 'Los Angeles').
state varchar State of the patient's address (e.g., 'NY', 'CA').
zip varchar Zip code of the patient's address (e.g., '10001', '90001').
language varchar Preferred language of the patient (e.g., 'English', 'Spanish').
race varchar Race of the patient (e.g., 'White', 'Black', 'Asian').
restype varchar Residence type of the patient (e.g., 'House', 'Apartment').
empName varchar Name of the patient's employer (e.g., 'Acme Corp', 'Tech Solutions').
empAddress1 varchar Street address line 1 of the patient's employer (e.g., '789 Market St', '321 Oak St').
empAddress2 varchar Street address line 2 of the patient's employer (e.g., 'Floor 3', 'Building 5').
empCity varchar City of the patient's employer's address (e.g., 'San Francisco', 'Chicago').
empState varchar State of the patient's employer's address (e.g., 'CA', 'IL').
empZipCode varchar Zip code of the patient's employer's address (e.g., '94103', '60601').
pharmacyid int Preferred pharmacy identifier, foreign key relation to the pharmacy's table (e.g., 11, 12, 13).
timeVal datetime Date and time when the record was created or updated (e.g., '2021-01-01 12:34:56').
ethnicity varchar Ethnicity of the patient (e.g., 'Hispanic', 'Non-Hispanic').
uparea varchar Area of the Patient(e.g., 'Floor 3', 'Building 5')
portalUpdateTimeVal datetime Date and time when the record was last updated through the patient portal (e.g., '2021-02-01 10:30:00').
