Table prisma_med_equipment_procedure_performer
Description This table stores information about the performers involved in medical equipment procedures, including their personal and organizational details.
Column DataType Description
id (Primary Key) bigint Unique identifier for each record in the table. Example: 1, 2, 3
procedureId bigint Identifier for the associated procedure. Foreign key relation to `prisma_med_equipment_procedure_entry` table's `id` column. Example: 101, 102, 103
userId varchar Identifier for the user performing the procedure. Example: "U123", "U124", "U125"
addressLine1 varchar First line of the performer's address. Example: "123 Main St", "456 Elm St"
addressLine2 varchar Second line of the performer's address (optional). Example: "Apt 4B", "Suite 200"
city varchar City where the performer is located. Example: "New York", "Los Angeles"
county varchar County where the performer is located. Example: "Los Angeles County", "Cook County"
state varchar State where the performer is located. Example: "CA", "NY", "TX"
postalcode varchar Postal code of the performer's address. Example: "90210", "10001"
country varchar Country where the performer is located. Example: "USA", "Canada"
homePhone varchar Performer's home phone number. Example: "(555) 123-4567", "(555) 987-6543"
workPhone varchar Performer's work phone number. Example: "(555) 234-5678", "(555) 876-5432"
faxnumber varchar Performer's fax number. Example: "(555) 345-6789", "(555) 765-4321"
personFName varchar Performer's first name. Example: "John", "Jane"
personLName varchar Performer's last name. Example: "Doe", "Smith"
personSufix varchar Performer's name suffix (optional). Example: "Jr.", "III"
organizationId varchar Identifier for the performer's organization. Example: "O123", "O124"
organizationName varchar Name of the performer's organization. Example: "MediCorp", "HealthTech"
organizationPhone varchar Phone number of the performer's organization. Example: "(555) 456-7890", "(555) 654-3210"
organizationFax varchar Fax number of the performer's organization. Example: "(555) 567-8901", "(555) 543-2109"
organizationAddLine1 varchar First line of the organization's address. Example: "789 Oak St", "321 Pine St"
organizationAddLine2 varchar Second line of the organization's address (optional). Example: "Floor 3", "Building B"
organizationCity varchar City where the organization is located. Example: "San Francisco", "Chicago"
organizationState varchar State where the organization is located. Example: "IL", "FL", "WA"
organizationpostalcode varchar Postal code of the organization's address. Example: "94102", "60601"
organizationCountry varchar Country where the organization is located. Example: "USA", "UK"
