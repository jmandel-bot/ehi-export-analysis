Table refund_check_print
Description This table stores information about refund checks that are printed, including details about the payee, check number, date, amount, memo lines, claims, and the user who printed the check.
Column DataType Description
id (Primary Key) int Unique identifier for each record in the table (e.g., 1, 2, 3).
refundid int Foreign key referring to the refund ID in refunds table (e.g., 101, 102, 103).
paytoname varchar Name of the payee to whom the check is issued (e.g., "John Doe", "Jane Smith").
checkno varchar Check number associated with the refund check (e.g., "123456", "789012").
checkDate varchar Date when the check was issued, in the format YYYY-MM-DD (e.g., "2021-01-01", "2021-12-31").
checkNumericAmount varchar Numeric amount of the check (e.g., "100.00", "250.50").
checkwordAmount varchar Amount of the check written in words (e.g., "One hundred dollars", "Two hundred fifty dollars and fifty cents").
checkMemoLine1 varchar First line of the memo on the check (e.g., "Refund for overpayment", "Claim reimbursement").
checkMemoLine2 varchar Second line of the memo on the check (optional).
checkMemoLine3 varchar Third line of the memo on the check (optional).
checkMemoLine4 varchar Fourth line of the memo on the check (optional).
checkRemitNotes text Additional notes related to the check remittance (e.g., "Processed on time", "Delayed due to system error").
claimNo1 varchar Claim numbers associated with the refund check (e.g., "C12345", "C67890").
claimAmt1 varchar Amounts associated with each claim (e.g., "50.00", "75.25").
claimNotes1 varchar Notes related to each claim (e.g., "Processed successfully", "Requires additional documentation").
claimNo10 varchar Claim numbers associated with the refund check (e.g., "C12345", "C67890").
claimAmt10 varchar Amounts associated with each claim (e.g., "50.00", "75.25").
claimNotes10 varchar Notes related to each claim (e.g., "Processed successfully", "Requires additional documentation").
printedById int ID of the user who printed the check (e.g., 1001, 1002, 1003).
printDate datetime Date when the check was printed, in the format YYYY-MM-DD (e.g., "2021-01-01", "2021-12-31").
printTime datetime Time when the check was printed, in the format HH:MM:SS (e.g., "09:00:00", "14:30:00").
deleteflag int Flag indicating if the record has been deleted (0 for not deleted, 1 for deleted).
claimPatientName1 varchar Names of the patients associated with each claim (e.g., "John Doe", "Jane Smith").
claimPatientName10 varchar Names of the patients associated with each claim (e.g., "John Doe", "Jane Smith").
claimDOS1 varchar Date of service for each claim, in the format YYYY-MM-DD (e.g., "2021-01-01", "2021-12-31").
claimDOS10 varchar Date of service for each claim, in the format YYYY-MM-DD (e.g., "2021-01-01", "2021-12-31").
accNumber varchar Account number associated with the refund check (e.g., "A123456", "A789012").
