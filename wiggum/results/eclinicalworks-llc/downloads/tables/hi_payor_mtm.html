EHI Export Schema
Table hi_payor_mtm
Description This table stores information about payor transactions related to medication therapy management (MTM) programs.
Column DataType Description
id (Primary Key) int Unique identifier for each record in the table. Example: 1, 2, 3.
patient_id int Unique identifier for the patient associated with the transaction. Example: 101, 102, 103.
txn_id varchar Unique identifier for the transaction. Example: "TXN123", "TXN124", "TXN125".
hi_txn_id varchar Unique identifier for the mtm create transaction created on healow insights cloud. Example: "HITXN123", "HITXN124", "HITXN125".
mtm_UID varchar Unique identifier for the medication therapy management program. Example: "MTMUID123", "MTMUID124", "MTMUID125".
mtm_dob datetime Date of birth of the patient associated with the MTM program. Example: "1990-01-01", "1985-05-15", "1978-12-31".
program_name varchar Name of the MTM program. Example: "Diabetes Management", "Asthma Control", "Hypertension Management". Default: NULL.
program_disclaimer varchar Disclaimer text associated with the MTM program. Example: "Results may vary", "Consult your doctor before starting the program". Stored as text.
program_description varchar Description of the MTM program. Example: "This program helps patients manage their diabetes", "This program assists patients in controlling their asthma". Stored as text.
payor_id varchar Unique identifier for the payor associated with the transaction. Example: "PAYOR123", "PAYOR124", "PAYOR125". Default: NULL.
payor_name varchar Name of the payor associated with the transaction. Example: "Insurance Company A", "Insurance Company B", "Insurance Company C".
notification_source varchar Source of the notification for the transaction. Example: "Email", "SMS", "Phone Call". Default: NULL.
provider_name varchar Name of the healthcare provider associated with the transaction. Example: "Dr. John Smith", "Dr. Jane Doe". Default: NULL.
provider_id int Unique identifier for the healthcare provider associated with the transaction. Example: 201, 202, 203. Default: NULL.
provider_npi varchar National Provider identifier (NPI) of the healthcare provider associated with the transaction. Example: "1234567890", "0987654321". Default: NULL.
pharmacist varchar Name of the pharmacist associated with the transaction. Example: "Pharmacist A", "Pharmacist B". Default: NULL.
pharmacist_phone varchar Phone number of the pharmacist associated with the transaction. Example: "555-123-4567", "555-987-6543". Default: NULL.
pharmacist_fax varchar Fax number of the pharmacist associated with the transaction. Example: "555-123-4568", "555-987-6544". Default: NULL.
pharmacist_review_date datetime Date of the pharmacist's review of the transaction. Example: "2021-01-01", "2021-02-15". Default: NULL.
updated_txn_id varchar Unique identifier for the updated transaction. Example: "UTXN123", "UTXN124", "UTXN125".
created_date datetime Date and time when the record was created. Example: "2021-01-01 12:34:56", "2021-02-15 10:30:45".
updated_date datetime Date and time when the record was last updated. Example: "2021-01-02 14:56:32", "2021-02-16 11:45:23".
deleteflag tinyint Flag to indicate if the record has been deleted. Example: 0 (not deleted), 1 (deleted). Default: 0.
