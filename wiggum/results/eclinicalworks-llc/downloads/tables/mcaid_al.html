Table mcaid_al
Description This table stores information about Medicaid claims, including patient details, insurance coverage, diagnosis, procedures, charges, and provider details.
Column DataType Description
ID (Primary Key) int Unique identifier for each record in the table (e.g., 1, 2, 3).
InvId int Invoice ID associated with the claim, can be NULL (e.g., 101, 102, 103).
MedicaidIDNo varchar Medicaid identification number of the patient, not NULL (e.g., '1234567890123').
FName varchar First name of the patient, not NULL (e.g., 'John', 'Jane').
LName varchar Last name of the patient, not NULL (e.g., 'Doe', 'Smith').
HICNo varchar Health insurance claim number, not NULL (e.g., '987654321098').
PatientAccountNo varchar Patient account number, not NULL (e.g., 'PA123456').
CoveredOtherIns char Indicates if the patient is covered by other insurance, can be NULL (e.g., 'Yes', 'No').
OtherIns varchar Name of the other insurance, can be NULL (e.g., 'Blue Cross Blue Shield').
InsCarrierCode varchar Insurance carrier code for the other insurance, can be NULL (e.g., 'BCBS123').
OtherInsAmt varchar Amount covered by the other insurance, can be NULL (e.g., '1000.00').
FirstDx varchar First diagnosis code, can be NULL (e.g., 'A1234').
SecondDx varchar Second diagnosis code, can be NULL (e.g., 'B2345').
ThirdDx varchar Third diagnosis code, can be NULL (e.g., 'C3456').
FourthDx varchar Fourth diagnosis code, can be NULL (e.g., 'D4567').
DatesOfServiceFrom1 varchar Start date of the first service, can be NULL (e.g., '2021-01-01').
DatesOfServiceThru1 varchar End date of the first service, can be NULL (e.g., '2021-01-31').
POS1 char Place of service code for the first service, can be NULL (e.g., '11').
ProcedureCode1 varchar Procedure code for the first service, can be NULL (e.g., '99213').
Mods1 varchar Modifiers for the first service, can be NULL (e.g., '25').
Units1 varchar Number of units for the first service, can be NULL (e.g., '1').
Charges1 varchar Charges for the first service, can be NULL (e.g., '150.00').
Allowed1 varchar Allowed amount for the first service, can be NULL (e.g., '120.00').
Coinsurance1 varchar Coinsurance amount for the first service, can be NULL (e.g., '10.00').
Deductible1 varchar Deductible amount for the first service, can be NULL (e.g., '5.00').
Paid1 varchar Paid amount for the first service, can be NULL (e.g., '105.00').
