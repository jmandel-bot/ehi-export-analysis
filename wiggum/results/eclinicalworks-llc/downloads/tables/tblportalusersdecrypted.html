Table tblportalusersdecrypted
Description This table stores information about portal users in a decrypted format.
Column DataType Description
UID int Unique identifier for the user (int). Example: 1, 2, 3.
uname varchar Username of the user (varchar). Example: "john_doe", "jane_smith".
upwd varchar User's password (varchar). Example: "password123", "mySecureP@ssw0rd".
ufname varchar User's first name (varchar). Example: "John", "Jane".
uminitial varchar User's middle initial (varchar). Example: "A", "B".
ulname varchar User's last name (varchar). Example: "Doe", "Smith".
uemail text User's email address (mediumtext). Example: "john.doe@example.com", "jane.smith@example.com".
uwebPage text User's personal web page URL (mediumtext). Example: "https://johndoe.com", "https://janesmith.com".
upaddress varchar User's primary address (varchar). Example: "123 Main St", "456 Elm St".
upcity varchar User's city (varchar). Example: "New York", "Los Angeles".
upstate varchar User's state (varchar). Example: "NY", "CA".
upPhone varchar User's phone number (varchar). Example: "555-123-4567", "555-987-6543".
zipcode varchar User's zip code (varchar). Example: "10001", "90001".
dob varchar User's date of birth (varchar). Example: "1990-01-01", "1985-12-31".
ssn varchar User's social security number (varchar). Example: "123-45-6789", "987-65-4321".
sex varchar User's gender (varchar). Example: "M", "F".
upaddress2 varchar User's secondary address (varchar). Example: "Apt 4B", "Suite 300".
suffix varchar User's name suffix (varchar). Example: "Jr", "III".
prefix varchar User's name prefix (varchar). Example: "Dr", "Ms".
initials varchar User's initials (varchar). Example: "J.D.", "J.S.".
delFlag int Flag to indicate if the user is deleted (int). Example: 0 (not deleted), 1 (deleted).
pmcid int Unique identifier for the user's PMC (int). Example: 1, 2, 3.
locked int Flag to indicate if the user's account is locked (int). Example: 0 (unlocked), 1 (locked).
FormsSigned tinyint Flag to indicate if the user has signed required forms (tinyint). Example: 0 (not signed), 1 (signed).
resetPassword tinyint Flag to indicate if the user needs to reset their password (tinyint). Example: 0 (no), 1 (yes).
vmid varchar Virtual machine identifier for the user (varchar). Example: "vm1", "vm2".
invalid tinyint Flag to indicate if the user's account is invalid (tinyint). Example: 0 (valid), 1 (invalid).
convDone tinyint Column not used anymore.
facilityid int Unique identifier for the user's associated facility (int). Example: 1, 2, 3.
emailSent int Flag to indicate if an email has been sent to the user (int). Example: 0 (not sent), 1 (sent).
activationdate datetime Date and time when the user's account was activated (datetime). Example: "2021-01-01 12:34:56".
oldSchemePwdReset varchar Old password reset scheme value (varchar). Example: "reset123", "reset456".
pwdGUID varchar Password reset GUID (varchar). Example: "a1b2c3d4-e5f6-a1b2-c3d4-e5f6a1b2c3d4".
pwdCreated datetime Date and time when the user's password was created (datetime). Example: "2021-01-01 12:34:56".
id (Primary Key) int The auto-incrementing primary key for the table (int). Example: 1, 2, 3.
isUnsubscribed int This flag indicates whether the user has unsubscribed from notifications (int) using a value of 0 (subscribed) or 1 (unsubscribed).
timeVal datetime Keeps track of the user's inserted date and time values (e.g., 2021-01-01 12:34:56).
portalUpdateTimeVal datetime The date and time of the last update to the patient's demographics, accessed from the Patient Portal (date and time). For example, 2021-01-01 12:34:56.
decryptedPreferredName varchar The user's preferred name in decrypted format (varchar). For example, Johnny, Janey, etc..
